# liberals wins?
country_event = {
	id = election.2
	title = election.2.t
	desc = election.2.d
	picture = GFX_report_event_fascist_gathering
	
	is_triggered_only = no

	trigger = {
		has_idea = elections_al
		liberalism > 0.5
		NOT = {has_government = liberalism }
	}
	
	option = {
		name = election.2.a
		ai_chance = {
			base = 0
			modifier = {
				add = 2
				can_lose_democracy_support = yes
			}
			modifier = {
				add = 1
				can_lose_unity = no
			}
		}
		set_politics = {
		ruling_party = liberalism
		elections_allowed = yes
		}
		set_country_flag = pro-liberalism
	}

	option = {
		name = election.2.b
		ai_chance = {
			base = 0
			modifier = {
				add = 10
				can_lose_unity = yes
			}
		}
		add_stability = -0.5
		set_country_flag = anti-liberalism
	}
}
